  // Search users by username
  async searchUsers(query: string) {
    const users = await prisma.user.findMany({
      where: {
        username: {
          contains: query,
          mode: 'insensitive',
        },
        deletedAt: null,
        enabled: true,
      },
      select: {
        id: true,
        username: true,
        profilePicture: true,
        gamesPlayed: true,
        gamesWon: true,
        points: true,
        position: true,
      },
      take: 20,
      orderBy: {
        points: 'desc',
      },
    });

    return users;
  }
